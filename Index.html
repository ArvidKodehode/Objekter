Oppgave 2: Legg til metoder

Metoder beskriver hva personen kan gjøre, for eksempel hilse, legge til en interesse, eller vise informasjon om seg selv.

const person = {
    // Andre egenskaper...
    hils() {
        return `Hei! Jeg heter ${this.navn}, er ${this.alder} år gammel og ${this.sivilstatus}.`;
    },
    leggTilInteresse(nyInteresse) {
        this.interesser.push(nyInteresse);
        return `Interessen "${nyInteresse}" er lagt til.`;
    },
    visInfo() {
        return \`
            Navn: ${this.navn}

            Alder: ${this.alder}

            Sivilstatus: ${this.sivilstatus}

            Interesser: ${this.interesser.join(", ")}

            Kroppsdeler: ${this.kroppsdeler.join(", ")}
        \`;
    }
};

Oppgave 3: Lag et program

Programmet lar brukeren velge en handling som objektet skal utføre.

function runMethod() {
    const inputMethod = document.getElementById("inputMethod").value;
    const outputDiv = document.getElementById("output");
    let result = "";

    if (inputMethod === "greet") {
        result = person.hils();
    } else if (inputMethod === "addInterest") {
        const nyInteresse = prompt("Skriv inn en ny interesse:");
        if (nyInteresse) {
            result = person.leggTilInteresse(nyInteresse);
        } else {
            result = "Ingen interesse ble lagt til.";
        }
    } else if (inputMethod === "showInfo") {
        result = person.visInfo();
    }

    outputDiv.innerHTML = result;
}
